% create a coloured box with arrow and title as cv section headline
% param 1: section title
%
\newcommand{\cvsection}[1]
{
    \vspace{4pt}
    {
        \textsc{\Large\expandafter\FirstLetter#1\relax}%
        \textsc{\large\expandafter\RestLetters#1\relax}%
    }\\[-5pt]
    \textcolor{black}{\hrule}
    \vspace{4pt}
}

% Commandes auxiliaires pour extraire la première lettre et le reste
\def\FirstLetter#1#2\relax{#1}
\def\RestLetters#1#2\relax{#2}

%
\newcommand{\metasection}[2]
{

}

%----------------------------------------------------------------------------------------
%	 CV EVENT

% creates a stretched box as cv entry headline followed by some paragraphs about 
% the work you did
% param 1:	event time i.e. 2014 or 2011-2014 etc.
% param 2:	event name (what did you do?)
% param 3:	institution (where did you work / study)
% param 4:	list of paragraphs

\newcommand{\cvevent}[4]
{ 
\begin{minipage}{\textwidth}  
    \hspace{0.5cm}  
    \begin{tabular*}{0.95\textwidth}{@{\extracolsep{\fill}}p{0.7\textwidth}r@{}}
        {\fontfamily{ptm}\selectfont\textbf{#2}} & #1 \\[2pt]
        {\fontfamily{ptm}\selectfont\small\textit{#3}} & \\[0pt]
    \end{tabular*}

    \hspace{1cm}  
    \begin{minipage}{0.9\textwidth}  
        \foreach \desc in {#4}{
            {\scalebox{0.85}{\textbullet}}\hspace{0.5em}\desc\\[2pt]
        }
    \end{minipage}
\end{minipage}
\vspace{0.2pt}
}

% creates a stretched box as 
\newcommand{\cveventmeta}[2]
{
	\mbox{\mystrut \hspace{87pt}\textit{#1}}\\
	#2
}
